def numProbability(v, mu, sigma):
    return (1.0/sqrt(2 * pi * (sigma ** 2))*exp(-((v-mu)**2)/(2*(sigma**2))))

def catProbability(sample,universe):
    return sample.shape[0]/universe.shape[0]
